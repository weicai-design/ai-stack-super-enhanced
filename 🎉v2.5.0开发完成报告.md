## 🌟 v2.5.0 特色功能

### 上下文记忆
```python
from ai.assistant.context_memory import get_context_memory

memory = get_context_memory()

# 记住对话
memory.add_conversation_turn(
    session_id="session_123",
    role="user",
    content="什么是AI？"
)

# 学习偏好
memory.learn_user_preference(
    user_id=123,
    action="search",
    context={"query": "AI"}
)

# 获取建议
suggestions = memory.get_personalized_suggestions(123, {})
```

### 工作流自动化
```python
from automation.workflow_engine import get_workflow_engine

engine = get_workflow_engine()

# 创建工作流
workflow = Workflow(name="数据处理", start_node_id="start")
engine.create_workflow(workflow)

# 添加任务节点
task = WorkflowNode(
    id="task1",
    type=NodeType.TASK,
    name="数据采集",
    config={"task_type": "log", "message": "采集数据..."}
)
engine.add_node(workflow.id, task)

# 执行工作流
execution = await engine.execute_workflow(workflow.id)
```

### 智能推荐
```python
from recommendation.hybrid_recommender import get_recommender

recommender = get_recommender()

# 添加交互数据
recommender.add_interaction(
    UserItem(user_id=1, item_id=1, rating=5.0)
)

# 获取推荐
recs = recommender.hybrid_recommend(
    user_id=1,
    top_k=10,
    cf_weight=0.6,  # 协同过滤权重
    cb_weight=0.4   # 基于内容权重
)
```

### 异常检测
```python
from prediction.anomaly_detection import get_anomaly_detector

detector = get_anomaly_detector()

# 检测异常
anomalies = detector.detect_all(value=200)

# 查看统计
stats = detector.get_statistics()

# 获取最近异常
recent = detector.get_recent_anomalies(count=10, severity="high")
```
